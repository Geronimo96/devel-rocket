
> 과제 1. 입력받은 횟수만큼 반복하는 기능 제작
- 사용자로부터 정수를 하나 입력받고, 그 수만큼 반복하는 문구 출력하기
  - "몇회 반복하시겠습니까?" 출력
  - 입력을 받기
  - "n 회 반복중입니다"을 반복적으로 출력. n에는 현재 반복회차 표시하기

### 답)
```csharp
using System;

namespace Loop_Practice
{
    internal class Program
    {
        static void Main(string[] args)
        {
            Console.Write("몇 회 반복하시겠습니까?: ");
            int inputNum = int.Parse(Console.ReadLine());

            for (int i = 1; i <= inputNum; i++)
            {
                Console.WriteLine($"{i}회 반복 중입니다.");                
            }          
        }
    }
}
```

> 과제 2. 구구단 기능 제작
- 유저로부터 정수 하나를 입력받고, 입력받은 수의 구구단을 출력하는 프로그램 제작하시오.
- 예를 들어 3을 입력 받으면 3단에 대한 구구단 출력. 단, 유저가 입력한 수가 9를 초과하거나 숫자가 아닌 것을 입력하면 정상적인 입력이 아닐 경우, 제대로 된 입력이 나올때까지 무한 반복하는 예외처리도 있어야 한다
"출력하고자 하는 구구단을 입력해주시길 바랍니다" 출력
  - 유저로부터 1~9까지의 숫자를 입력받음
  - "3x1 = 3, 3x2 =6, 3x3 = 9" 등등 해당 구구단 출력
  - 만약 1~9가 아닌 숫자 혹은 문자열이 들어오면 다시 입력하라고 반복시키기

### 답)
```csharp
uusing System;

namespace Loop_Practice
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int dan = 0;
            bool isValidInput = false;

            //올바른 수 ( 1~9인 정수) 입력 받을 때까지 반복
            while (!isValidInput) 
            {
                Console.Write("출력하고자 하는 구구단을 입력해주시길 바랍니다 (1~9 사이의 정수):");
                string inputValue = Console.ReadLine(); // 사용자 입력 받기

                if (int.TryParse(inputValue, out dan)) // 예외 처리 
                {
                   if (dan >= 1 && dan <= 9) //변환된 정수가 1~9 숫자인지 
                    {
                        isValidInput = true; //올바른 수인 경우 루프 종료    
                    }
                   else
                    {   // 9를 초과하거나 1 미만인 경우
                        Console.WriteLine($"!에러: {dan}은(는) 1부터 9까지의 숫자가 아닙니다. 다시 입력해주세요.");
                    }
                }
                else
                {   // int 형으로 변환이 안되는 경우
                    Console.WriteLine("유효한 숫자가 아닙니다. 다시 입력해주세요.");
                }
            }

            //구구단 출력
            Console.WriteLine($"--------{dan}단---------");
            for (int i = 1; i <= 9; i++)
            {
                Console.WriteLine($"{dan} x {i} = {dan * i}");
            }

            Console.WriteLine("---------------------");

        }
    }
}


```

> 과제 3. 별 찍기
  - 중첩 반복문을 활용하여 별 기호(*)로 삼각형 패턴을 출력하는 프로그램을 작성하시오.

### 왼쪽 정렬 직각 삼각형
```csharp
using System;

public class LeftTriangle
{
    public static void Main(string[] args)
    {
        int height = 5; // 삼각형의 높이 설정

        Console.WriteLine($"--- 왼쪽 정렬 직각 삼각형 (높이: {height}) ---");
        
        // i: 바깥쪽 루프 (행, Row) 제어, 1부터 height까지 증가
        for (int i = 1; i <= height; i++)
        {
            // j: 안쪽 루프 (별, Star) 제어
            // 현재 행 번호(i)만큼 별을 출력
            for (int j = 1; j <= i; j++)
            {
                Console.Write("*");
            }

            // 한 행의 출력이 끝난 후 줄 바꿈
            Console.WriteLine();
        }
    }
}
```

### 오른쪽 정렬 직각 삼각형

```csharp
using System;

public class RightTriangle
{
    public static void Main(string[] args)
    {
        int height = 5; // 삼각형의 높이 설정

        Console.WriteLine($"--- 오른쪽 정렬 직각 삼각형 (높이: {height}) ---");

        // i: 바깥쪽 루프 (행, Row) 제어, 1부터 height까지 증가
        for (int i = 1; i <= height; i++)
        {
            // 1. 공백 출력 루프 (j):
            // 필요한 공백의 개수는 (높이 - 현재 행 번호) 입니다. (예: 5 - 1 = 4개의 공백)
            for (int j = 1; j <= height - i; j++)
            {
                Console.Write(" ");
            }

            // 2. 별 출력 루프 (k):
            // 필요한 별의 개수는 현재 행 번호(i)와 같습니다.
            for (int k = 1; k <= i; k++)
            {
                Console.Write("*");
            }

            // 한 행의 출력이 끝난 후 줄 바꿈
            Console.WriteLine();
        }
    }
}
```
